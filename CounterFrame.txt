    private void count_btnActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_count_btnActionPerformed
        int valueOfIncrement = Integer.parseInt(countInput.getText());
    Thread thread = new Thread(() -> {
        for (int i = 0; i < valueOfIncrement; i++) {
            atomicCounter.incrementAndGet();
            synchronized (syncObject) {
                synCounter++;
            }
            int finalValue = i; // Capture the current value of i for use in the UI update
            java.awt.EventQueue.invokeLater(() -> {
                countOutput.setText(String.valueOf(finalValue));
            });
        }
    });

    thread.start();
}
